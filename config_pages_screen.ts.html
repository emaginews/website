/**
 * 스크린 클래스
 * @date 8/24/2023 - 3:42:45 AM
 *
 * @export
 * @interface Screen
 * @typedef {Screen}
 * @template T 스크린 타입
 * @template S 스크린 분류
 */
export interface Screen<T extends string, S extends string> {
  /**
   * 스크린 id
   * @readonly
   * @type {?string}
   */
  readonly id?: string;

  /**
   * 스크린 명
   * @readonly
   * @type {?string}
   */
  readonly name?: string;

  /**
   * 스크린 타입
   * @readonly
   * @type {T}
   */
  readonly type: T;

  /**
   * 스크린 분류
   * @readonly
   * @type {S}
   */
  readonly category: S;
}

export interface Screens<T extends string, S extends string> {
  [key: string]: Screen<T, S>;
}

/**
 * Description placeholder
 * @export
 * @template T 스크린 타입
 * @template S 스크린 분류
 * @param {string} screenId 스크린 id
 * @param {T} type 스크린 타입
 * @param {S} category 스크린 분류
 * @param {?string} [name] 스크린명
 * @returns {Screen<T, S>} 스크린 클래스
 */
export function createScreen<T extends string, S extends string>(
  screenId: string,
  type: T,
  category: S,
  name?: string
): Screen<T, S> {
  return {
    id: screenId,
    name,
    type,
    category
  };
}
