/**

 * @desc 네이티브(ios) 스키마 호출 인터페이스 adapter
 * @package @linker/web-front

 * @version 0.1.0
 * @see =================== 변경 내역 ==================
 * <pre>
 * [작성자][작업일시] - 내용
 * [wskim][2023.08.22 Tuesday 16:36:52] - 최초작성
 * </pre>
 */
import _ from 'lodash';
import { JsonString } from '../interface.type';

/**
 * URL scheme 방식 네이티브 인터페이스 호출 adapter
 * @date 8/24/2023 - 12:20:09 AM
 *
 * @export
 * @param {string} schemeUrl scheme URL
 * @param {string} action 인터페이스 명
 * @param {?JsonString} [actionArgs] 인터페이스 호출 파라미터
 */
export function nativeSchemeInterfaceAdapter(
  schemeUrl: string,
  action: string,
  actionArgs?: JsonString
) {
  const queryString = actionArgsToUrlQuery(actionArgs);
  const uri = `${schemeUrl}${queryString && `?${queryString}`}`;

  location.href = uri;
}

/**
 * 네이티브 인터페이스 파라미터를 url 쿼리 스트링으로 변환하는 함수
 * (단순한 JSON(또는 Map) 형식을 기반으로 규약하며, 재귀적으로 처리하지 않음)
 * (배열 등 field 값이 복잡한 경우는 JSON 문자열 전달을 기본으로 함)
 * @date 8/24/2023 - 12:20:09 AM
 *
 * @param {?JsonString} [actionArgs] 인터페이스 호출 파라미터
 * @returns {*} url 쿼리스트링
 */
function actionArgsToUrlQuery(actionArgs?: JsonString) {
  if (!actionArgs) {
    return '';
  }
  const args = JSON.parse(actionArgs);

  // prettier-ignore
  const query = _.reduce(args, (acc, val, key) => {
    const value = _.isObject(val) ? JSON.stringify(val) : val;
    const param = `${key}=${value}`;
    if (!acc) {
      acc += `${param}`;
    } else {
      acc += `&${param}`;
    }
    return acc;
  }, '');

  return query;
}
