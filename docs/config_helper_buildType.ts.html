 * [wskim][2023.08.22 Tuesday 15:46:05] - 최초작성
 * </pre>
 */
import _ from 'lodash';

/**
 * 빌드 타입 ENUM 객체
 * @date 8/24/2023 - 2:06:52 AM
 *
 * @export
 * @enum {number}
 */
export enum BuildEnv {
  LOCAL = 'local',
  DEV = 'dev',
  STG = 'stg',
  PRD = 'prd'
}

/**
 * 빌드 타입
 * @date 8/24/2023 - 2:06:52 AM
 *
 * @export
 * @typedef {BuildEnvStr}
 */
export type BuildEnvStr = keyof typeof BuildEnv;

/**
 * BuildEnv 객체 변환 함수
 * @date 8/24/2023 - 2:06:52 AM
 *
 * @export
 * @param {?string} [type] 빌드타입 문자열
 * @returns {BuildEnv} BuildEnv 객체
 */
export function toBuildEnv(type?: string) {
  const _type = type?.toUpperCase();
  if (_type && _.has(BuildEnv, _type)) {
    return BuildEnv[_type as BuildEnvStr];
  }
  return BuildEnv.DEV;
}

/**
 * 개발환경 여부 확인 함수
 * @date 8/24/2023 - 2:06:52 AM
 *
 * @export
 * @param {BuildEnv} env 빌드환경
 * @returns {boolean} 개발환경 여부
 */
export function isDev(env: BuildEnv) {
  return !(env === BuildEnv.PRD || env === BuildEnv.STG);
}
