import { createScreen, Screen } from './screen';

/**
 * URL Path 타입
 * @typedef {PathName}
 */
type PathName = `/${string}`;

/**
 * 페이지 클래스
 * @date 8/24/2023 - 2:09:04 AM
 *
 * @export
 * @interface Page
 * @template T 페이지 타입(일반, 팝업, alert 등)
 * @template S URL Path 타입
 * @template U 권한 타입
 * @extends {Screen<T, S>}
 */
export interface Page<T extends string, S extends PathName, U>
  extends Screen<T, S> {
  /**
   * 페이지 URL path
   * @readonly
   * @type {S}
   */
  readonly pathname: S;

  /**
   * 권한 정보
   * @readonly
   * @type {?U}
   */
  readonly permission?: U;

  /**
   * 페이지 title
   * @type {?string}
   */
  title?: string;

  // isPageLoading?: boolean;
}

/**
 * 페이지 정보 생성 함수
 * @export
 * @template T 페이지 타입(일반, 팝업, alert 등)
 * @template S 페이지 URL path
 * @template U 페이지 권한
 * @param {object} props
 * @returns {Page<T, S, U>} 페이지 정보
 */
export function createPage<T extends string, S extends PathName, U>(props: {
  screenId?: string; // 페이지(스크린) id
  screenType: T; // 페이지 타입(일반, 팝업, alert 등)
  name?: string; // 페이지명
  title?: string; // 페이지 헤더명
  category?: S; // 페이지 분류
  permission?: U; // 페이지 권한
  pathname: S; // 페이지 url path
}) {
  const {
    name,
    title,
    screenType,
    permission,
    pathname,
    screenId = pathname,
    category = pathname
  } = props;
  const screen = createScreen(screenId, screenType, category, name);
  const page: Page<T, S, U> = {
    ...screen,
    name,
    permission,
    title: title || name,
    pathname: pathname || `/${screen.id}`
  };
  return page;
}
