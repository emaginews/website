/**

 * @export
 * @typedef {Platform}
 */
export type Platform = 'ios' | 'aos' | 'pc' | 'browser';

/**
 * 네이티브 인터페이스 호출 base 파라미터
 * @date 8/24/2023 - 12:21:10 AM
 *
 * @export
 * @interface BaseNativeInterfaceParams
 * @typedef {BaseNativeInterfaceParams}
 */
export interface BaseNativeInterfaceParams {
  /**
   * 콜백 id
   * @date 8/24/2023 - 12:21:10 AM
   *
   * @type {?number}
   */
  callbackId?: number;
}

/**
 * 네이티브 인터페이스
 * @date 8/24/2023 - 12:21:10 AM
 *
 * @export
 * @interface NativeInterfaceAction
 * @typedef {NativeInterfaceAction}
 * @template Action 액션 타입
 * @template Params 파라미터 타입
 */
export interface NativeInterfaceAction<Action extends string, Params = any> {
  /**
   * 액션
   * @date 8/24/2023 - 12:21:10 AM
   *
   * @type {Action}
   */
  action: Action;

  /**
   * 파라미터
   * @date 8/24/2023 - 12:21:10 AM
   *
   * @type {?Params}
   */
  actionParams?: Params;
}

/**
 * 네이티브 인터페이스 콜백(일반)
 * @date 8/24/2023 - 12:21:10 AM
 *
 * @export
 * @typedef {Callback}
 * @template T 콜백 response 타입
 */
export type Callback<T> = (res: T) => void;

/**
 * 네이티브 인터페이스 성공 콜백
 * @date 8/24/2023 - 12:21:09 AM
 *
 * @export
 * @typedef {JSInterfaceSuccessCallback}
 * @template T 콜백 response 타입
 */
export type JSInterfaceSuccessCallback<T> = Callback<T>;

/**
 * 네이티브 인터페이스 실패 콜백
 * @date 8/24/2023 - 12:21:09 AM
 *
 * @export
 * @typedef {JSInterfaceFailCallback}
 */
export type JSInterfaceFailCallback = Callback<string>;

 *
 * @export

export type JsonString = string;

/**
 * 웹-앱 연동 인터페이스
 * @date 8/24/2023 - 12:21:09 AM
 *
 * @export
 * @typedef {IJSInterface}
 */
export type IJSInterface = {
  platform: () => Platform;

  nativeCallback: (
    callbackId: number,
    args: JsonString,
    isSuccess: boolean,
    keepCallback: boolean
  ) => void;

  dispatchEvent: (eventName: string, data?: any) => void;

  execNativeInterface: <
    Action extends string,
    Params,
    CallbackData,
    T extends NativeInterfaceAction<Action, Params>
  >(
    action: T['action'],
    actionArgs?: T['actionParams'],
    successCallback?: JSInterfaceSuccessCallback<CallbackData>,
    failCallback?: JSInterfaceFailCallback
  ) => void;

  _execNativeInterface: <T, S = any>(
    action: string,
    actionArgs?: T,
    successCallback?: JSInterfaceSuccessCallback<S>,
    failCallback?: JSInterfaceFailCallback
  ) => void;

  _setCallback: <S = any>(
    successCallback?: JSInterfaceSuccessCallback<S>,
    failCallback?: JSInterfaceFailCallback
  ) => number | undefined;
};

/**
 * 앱-웹 연동 성공/실패 콜백
 * @date 8/24/2023 - 12:21:09 AM
 *
 * @export
 * @typedef {CallbackRef} 앱-웹 연동 성공/실패 콜백
 */
export type CallbackRef = {
  success?: JSInterfaceSuccessCallback<any>;
  fail?: JSInterfaceFailCallback;
};
