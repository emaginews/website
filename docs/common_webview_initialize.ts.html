/**

 * @desc (웹뷰 통신을 위한) 웹-앱 인터페이스 초기 설정
 * @package @linker/web-front

 * @since 2023.08.22 19:43:01
 * @version 0.1.0
 * @see =================== 변경 내역 ==================
 * <pre>
 * [작성자][작업일시] - 내용
 * [wskim][2023.08.22 Tuesday 19:43:01] - 최초작성
 * </pre>
 */
import assert from 'assert';
import { Platform, IJSInterface } from './interface.type';
import { isNative } from './helper/getPlatform';
import { JSInterface } from './interface';
import { nativeSchemeInterfaceAdapter } from './impl/schemeInteface.adapter';
import { nativeBridgeInterfaceCallee } from './impl/bridgeInterface.callee';

/**
 * 인터페이스 설정 초기화(브라우저 환경에서는 인터페이스 설정하지 않음)
 * @date 8/24/2023 - 12:20:29 AM
 *
 * @export
 * @param {Platform} platform 플랫폼 정보
 * @param {boolean} useInternalScheme 내부 앱스킴 사용여부
 * @param {?string} [schemeUrl] 내부 앱스킴 url
 * @param {IJSInterface} [jsInterface=new JSInterface()] 웹(자바스크립트) 인터페이스
 * @param {*} [scope=window] 인터페이스 scope(기본: 전역 window 객체)
 */
export function initInterface(
  platform: Platform,
  useInternalScheme: boolean,
  schemeUrl?: string,
  jsInterface: IJSInterface = new JSInterface(),
  scope: any = window
) {
  _initJavascriptInterface(platform, jsInterface, scope);
  _initNativeInterface(platform, useInternalScheme, schemeUrl, scope);
}

/**
 * Description placeholder
 * @date 8/24/2023 - 12:20:29 AM
 *
 * @export
 * @param {Platform} platform 웹(자바스크립트) 인터페이스 초기화(private 함수)
 * @param {IJSInterface} jsInterface 웹(자바스크립트) 인터페이스
 * @param {*} [scope=window] 인터페이스 scope(기본: 전역 window)
 */
export function _initJavascriptInterface(
  platform: Platform,
  jsInterface: IJSInterface,
  scope: any = window
) {
  assert.ok(scope instanceof Object);

  if (isNative(platform) && !scope.hasOwnProperty('web')) {
    console.log('initialize web interface');
    scope.web = jsInterface;
  }
}

/**
 * 네이티브 인터페이스 초기화(private 함수)
 * @date 8/24/2023 - 12:20:29 AM
 *
 * @export
 * @param {Platform} platform 플랫폼 정보
 * @param {boolean} useInternalScheme 내부 앱스킴 사용여부
 * @param {?string} [schemeUrl] 내부 앱스킴 url
 * @param {*} [scope=window] 인터페이스 scope(기본: 전역 window)
 */
export function _initNativeInterface(
  platform: Platform,
  useInternalScheme: boolean,
  schemeUrl?: string,
  scope: any = window
) {
  if (isNative(platform)) {
    const initNativeAdapter = (plt: Platform, useScheme?: boolean) => {
      if (useScheme) {
        if (!scope.hasOwnProperty('mob')) {
          console.log(`scheme url: initialize native(${plt}) interface`);
          // eslint-disable-next-line no-undef
          const adapter: NativeInterface = nativeSchemeInterfaceAdapter.bind(
            null,
            schemeUrl || ''
          );
          scope.mob = {
            interface: adapter
          };
        }
      } else {
        // linker 앱에서만 정상구동(모바일 웹, 타 앱 웹뷰 연동시 오류 발생)
        // assert.ok(scope.hasOwnProperty('mob') && !scope.hasOwnProperty(plt));
        console.log(`bridge callee: initialize native(${plt}) interface`);
        scope[plt] = {
          interface: nativeBridgeInterfaceCallee
        };
      }
    };

    // 디바이스에 관계없이 인터페이스 구조 통일을 위해 url 스키마 인터페이스 어댑터 설정
    // 기본(ios: 스키마 방식, aos: 브릿지 방식) <- .env 환경변수 파일 참조
    if (platform === 'ios') {
      initNativeAdapter(platform, useInternalScheme);
    }
    if (platform === 'aos') {
      initNativeAdapter(platform, useInternalScheme);
    }
  }
}
