/**
 * @title pages.ts
 * @desc 웹 페이지 목록 정보
 * @package @linker/web-front
 *
 * @author linker
 * @since 2023.08.23 Wednesday 00:39:08
 * @version 0.1.0
 * @see =================== 변경 내역 ==================
 * <pre>
 * [작성자][작업일시] - 내용
 * [wskim][2023.08.23 Wednesday 00:39:08] - 최초작성
 * </pre>
 */
import { createPage } from './page';

/**
 * 기본 페이지 권한 타입(필요시 변경)
 * @export
 * @typedef {PermissionCd}
 */
export type PermissionCd =
  | '0' // 비회원 or 미로그인
  | '1' // 준회원;
  | '2'; // 정회원;

/**
 * 기본 스크린 타입(필요시 변경)
 * @export
 * @typedef {LinkerScreenType}
 */
export type LinkerScreenType =
  | 'page'
  | 'tab'
  | 'popup-full'
  | 'popup-bottom'
  | 'popup-drag'
  | 'confirm'
  | 'prompt'
  | 'alert';

/**
 * 기본 페이지 URL Path 타입(필요시 변경)
 * @export
 * @typedef {LinkerPageNamespace}
 */
export type LinkerPageNamespace =
  | '/pagelisst'
  | '/notfound'
  | '/error'
  | '/main/page1'
  | '/main/page2'
  | '/main/page3';

/**
 * 페이지 정보 생성 generic 함수(필요시 generic 타입 변경)
 * @type {Page<LinkerScreenType, LinkerPageNamespace, PermissionCd>}
 */
const defaultPageInfoProvider = createPage<
  LinkerScreenType,
  LinkerPageNamespace,
  PermissionCd
>;
const pageInfo = defaultPageInfoProvider;

/**
 * 기본 페이지 타입(필요시 변경)
 * @export
 * @typedef {LinkerPage}
 */
export type LinkerPage = ReturnType<typeof pageInfo>;

const pageList = pageInfo({ screenType: 'page', pathname: '/pagelisst' });
const notFound = pageInfo({ screenType: 'page', pathname: '/notfound' });
const errPage = pageInfo({ screenType: 'page', pathname: '/error' });

/**
 * 페이지 목록 정보
 * @type {LinkerPage[]}
 */
export const pages = [pageList, errPage, notFound];
