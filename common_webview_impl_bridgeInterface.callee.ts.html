import _ from 'lodash';
import { JsonString } from '../interface.type';

/**
 * Android 자바스크립트 브릿지 함수 호출
 * @date 8/24/2023 - 12:19:38 AM
 *
 * @export
 * @param {string} action 인터페이스 명
 * @param {?JsonString} [actionArgs] 인터페이스 파라미터
 */
export function nativeBridgeInterfaceCallee(
  action: string,
  actionArgs?: JsonString
) {
  const bridge = getJavascriptBridgeWrapper(action);
  if (bridge) {
    bridge(actionArgs);
  } else {
    console.log(`invalid native interface: ${action}`);
  }
}

/**
 * 네이티브 브릿지 메서드 호출 wrapper
 * 직접 호출시 오류 발생: Java bridge method can't be invoked on a non-injected object
 * @date 8/24/2023 - 12:19:38 AM
 *
 * @param {string} action 인터페이스 명
 * @returns {*}
 */
function getJavascriptBridgeWrapper(action: string) {
  const hasJsBridge = window.mob && _.has(window.mob, action);
  if (hasJsBridge) {
    // eslint-disable-next-line no-undef
    const wrapper: BridgeInterface = function (actionArgs?: JsonString) {
      if (window.mob && _.has(window.mob, action)) {
        window.mob[action](actionArgs);
      }
      console.log(`no js bridge: ${action}`);
    };
    return wrapper;
  }
  return undefined;
}
