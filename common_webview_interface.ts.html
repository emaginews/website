/**
 * @title interface.ts
 * @desc (웹뷰 통신을 위한) 웹-앱 연동 인터페이스
 * 웹에서 execNativeInterface 메서드를 호출하여 네이티브 인터페이스 실행
 * 네이티브에서는 nativeCallback 메서드를 호출하여 자바스크립트 콜백 함수(웹에서 전달받은 콜백 함수 id) 제어
 * @package @linker/web-front
 *
 * @author linker
 * @since 2023.08.22 16:45:24
 * @version 0.1.0
 * @see =================== 변경 내역 ==================
 * <pre>
 * [작성자][작업일시] - 내용
 * [wskim][2023.08.22 Tuesday 16:45:24] - 최초작성
 * </pre>
 */
import _ from 'lodash';
import { getPlatform } from './helper/getPlatform';
import { executeNtivINF } from './impl/execute';
import {
  CallbackRef,
  IJSInterface,
  JSInterfaceSuccessCallback,
  JSInterfaceFailCallback,
  JsonString,
  NativeInterfaceAction
} from './interface.type';
import {
  dispatchJsEvent,
  onInterfaceErrorCallback
} from './helper/eventHandler';

/**
 * 웹-앱 연동 인터페이스 클래스
 * @date 8/24/2023 - 12:21:03 AM
 *
 * @export
 * @class JSInterface
 * @implements {IJSInterface}
 */
export class JSInterface implements IJSInterface {
  /**
   * 콜백 id
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @private
   * @type {number} 콜백 id
   */
  private _callbackId: number;

  /**
   * 콜백함수 참조 배열
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @private
   * @type {CallbackRef[]} 콜백함수 참조 배열
   */
  private _callbacks: CallbackRef[] = [];

  /**
   * Creates an instance of JSInterface.
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @constructor
   */
  constructor() {
    this._callbackId = Math.floor(Math.random() * 2000000000);
  }

  /**
   * 플랫폼 getter
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @public
   * @returns {Platform} 플랫폼 정보
   */
  public platform() {
    return getPlatform();
  }

  /**
   * 웹제공 인터페이스 함수
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @public
   * @param {string} name 인터페이스 명
   * @param {JsonString} data 인터페이스 호출 파라미터(JSON 문자열로 serialize)
   */
  public interface(name: string, data: JsonString) {
    // @todo safe json parse
    // const json = JSON.parse(data);

    switch (name) {
      // case 'onGoBack':
      //   // @todo 이벤트 전달 > useEffect 훅에서 useOnGoBack 처리
      //   break;
      case 'onNativeError':
        onInterfaceErrorCallback(data);
        break;
      // case 'onDispatchEvent':
      //   this.dispatchEvent(json.eventName, json.data);
      //   break;
      default:
        console.log(`invalid js interface: ${name}`);
        break;
    }
  }

  /**
   * 네이티브에서 웹으로 이벤트 전달 함수
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @public
   * @param {string} eventName 이벤트 명
   * @param {?*} [data] 이벤트 데이터
   */
  public dispatchEvent(eventName: string, data?: any) {
    dispatchJsEvent(eventName, data);
  }

  /**
   * 네이티브에서 호출하는 자바스크립트 콜백 함수
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @public
   * @param {number} callbackId 콜백 id
   * @param {JsonString} args 콜백 데이터(JSON 문자열)
   * @param {boolean} [isSuccess=true] 인터페이스 호출 성공 여부
   * @param {boolean} [keepCallback=true] 콜백 유지 여부
   */
  public nativeCallback(
    callbackId: number,
    args: JsonString,
    isSuccess = true,
    keepCallback = true
  ) {
    const callback = this._callbacks[callbackId];
    if (callback) {
      const parsedArgs = JSON.parse(args);
      if (isSuccess) {
        if (callback.success) {
          callback.success(parsedArgs);
        }
      } else {
        // eslint-disable-next-line no-lonely-if
        if (callback.fail) {
          callback.fail(parsedArgs);
        }
      }
    }

    if (!keepCallback) {
      delete this._callbacks[callbackId];
    }
  }

  /**
   * 네이티브 인터페이스 실행 메서드(strict) - 권장(action, actionParams 타입 일치 여부 체크)
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @public
   * @template Action 네이티브 인터페이스 액션 타입
   * @template Params 네이티브 인터페이스 파라미터 타입
   * @template CallbackData 성공 콜백 함수 response 타입
   * @template T 네이티브 인터페이스 타입
   * @param {T['action']} action 네이티브 인터페이스 명
   * @param {?T['actionParams']} [actionArgs] 네이티브 인터페이스 파라미터
   * @param {?JSInterfaceSuccessCallback<CallbackData>} [successCallback] 성공 콜백 함수
   * @param {JSInterfaceFailCallback} [failCallback=onInterfaceErrorCallback] 실패 콜백 함수
   */
  public execNativeInterface<
    Action extends string,
    Params,
    CallbackData,
    T extends NativeInterfaceAction<Action, Params> = NativeInterfaceAction<
      Action,
      Params
    >
  >(
    action: T['action'],
    actionArgs?: T['actionParams'],
    successCallback?: JSInterfaceSuccessCallback<CallbackData>,
    failCallback: JSInterfaceFailCallback = onInterfaceErrorCallback
  ) {
    this._execNativeInterface(
      action,
      actionArgs,
      successCallback,
      failCallback
    );
  }

  /**
   * Description placeholder
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @public
   * @template T 네이티브 인터페이스 파라미터 타입
   * @template S 성공 콜백 함수 response 타입
   * @param {string} action 네이티브 인터페이스 명
   * @param {?T} [actionArgs] 네이티브 인터페이스 파라미터
   * @param {?JSInterfaceSuccessCallback<S>} [successCallback] 성공 콜백 함수
   * @param {JSInterfaceFailCallback} [failCallback=onInterfaceErrorCallback] 실패 콜백 함수
   */
  public _execNativeInterface<T, S = any>(
    action: string,
    actionArgs?: T,
    successCallback?: JSInterfaceSuccessCallback<S>,
    failCallback: JSInterfaceFailCallback = onInterfaceErrorCallback
  ) {
    const callbackId = this._setCallback(successCallback, failCallback);
    executeNtivINF(action, actionArgs, callbackId);
  }

  /**
   * 콜백함수 설정 함수
   * @date 8/24/2023 - 12:21:03 AM
   *
   * @public
   * @template S 성공 콜백 함수 response 타입
   * @param {?JSInterfaceSuccessCallback<S>} [successCallback] 성공 콜백 함수
   * @param {JSInterfaceFailCallback} [failCallback=onInterfaceErrorCallback] 실패 콜백 함수
   * @returns {*}
   */
  public _setCallback<S = any>(
    successCallback?: JSInterfaceSuccessCallback<S>,
    failCallback: JSInterfaceFailCallback = onInterfaceErrorCallback
  ) {
    let callbackId = undefined;
    if (successCallback || failCallback) {
      callbackId = this._callbackId;
      this._callbackId += 1;
      this._callbacks[callbackId] = {
        success: successCallback,
        fail: failCallback
      };
    }
    return callbackId;
  }
}
